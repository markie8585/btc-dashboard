<script>
async function getBTCPrice() {
    try {
        const response = await fetch("https://api.coingecko.com/api/v3/simple/price?ids=bitcoin&vs_currencies=eur");
        const data = await response.json();

        const price = data.bitcoin.eur;

        document.getElementById("btc-price").innerText = "€ " + price.toLocaleString();

        const now = new Date();
        document.getElementById("update-time").innerText =
            "Laatste update: " + now.toLocaleTimeString();

    } catch (error) {
        document.getElementById("btc-price").innerText = "Fout bij laden";
    }
}

// elke 10 seconden refresh
getBTCPrice();
setInterval(getBTCPrice, 10000);
</script>
